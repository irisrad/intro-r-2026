Transforming Values

String extraction = pull digits or patterns from text.
Conversion = turn extracted text into numbers.
Text analysis (intro) = tokenize words and count with tidytext and stringr.

Extract digits (stringr)

Code 
survey <- survey |>
  dplyr::mutate(
    commute_digits  = stringr::str_extract(commute_time, "[0-9]+"),
    distance_digits = stringr::str_extract(distance,     "[0-9]+")
  )

Ensure optional columns exist (safe if optional step skipped)

Code 
if (!"commute_words_num" %in% names(survey))  survey <- dplyr::mutate(survey,  commute_words_num  = NA_character_)
if (!"distance_words_num" %in% names(survey)) survey <- dplyr::mutate(survey, distance_words_num = NA_character_)

Coerce to numeric

Code 
survey <- survey |>
  dplyr::mutate(
    commute_min = as.numeric(dplyr::coalesce(commute_digits,  commute_words_num)),
    distance_mi = as.numeric(dplyr::coalesce(distance_digits, distance_words_num))
  )

Tidy text basics (tidytext + stringr)

Code 
# tidytext is loaded from training_source.R
# Tokens = words from r_hope; lowercase; remove stop words
hope_tokens <- survey |>
  dplyr::select(r_hope) |>
  dplyr::mutate(r_hope = tolower(coalesce(r_hope, ""))) |>
  tidytext::unnest_tokens(word, r_hope) |>
  dplyr::anti_join(tidytext::stop_words, by = c("word" = "word")) |>
  dplyr::filter(word != "")

# Top words
hope_top <- hope_tokens |>
